#include <stdio.h>

int main()
{
    int r[10][10];
    int i, j, a, q;
    q = 0;

    for(i = 0; i < 10; ++i)
    {
    for(j = 0; j < 10; ++j)
        {
       r[i][j] = 0;
        
        }
    }
    
    while((a = getchar()) != EOF)
    {
            if(a >= '0' && a <= '9') /*Enter the loop and poop if its a number*/
            {
                if(r[q][a - '0'] == a - '0') /* if the array space already equals the number*/
                {
                    while((r[q][a - '0'] == a - '0'))
                    {
                        ++q;
                    }
                    r[q][a - '0'] = a - '0';
                    q = 0;
                }
                else /* if the array space doesnt already equals the number*/
                {
                r[q][a - '0'] = a -'0';
                }
            }    
    }
    
for(i = 0; i < 10; ++i)
    {
    for(j = 0; j < 10; ++j)
        {
      printf("%d\t", r[i][j]);
        }
       printf("\n");
    }
    return 0;
}
